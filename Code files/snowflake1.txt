import tkinter as tk
import turtle

root = tk.Tk()

turtle.speed("slow")
turtle.pencolor("white")
turtle.pensize(10)

turtle.Screen().bgcolor("red")

def vsnow():
    turtle.right(30)
    turtle.forward(60)
    turtle.backward(60)
    turtle.left(60)

    turtle.forward(60)
    turtle.backward(60)
    turtle.right(30)


vsnow()

for i in range(0,4):
    turtle.forward(30)
    vsnow()
    turtle.backward(120)


    






root.mainloop()
